
plugin.tx_solr {

	index {

		files = 1
		files {
				// defines which files types to index. Comma separated list of file type suffixes, example: pdf, doc, docx
			allowedTypes = *

			duplicateCheck {
				operator = AND

				useFileChecksum = 0
				useFileName = 1
				useFilePath = 1
				useLanguage = 0
				useReferenceDocumentId = 1
			}
		}

		queue.tx_solr_file.fields {

		}

	}

	search {

		results {

			fieldRenderingInstructions {

				url = CASE
				url {
					tx_solr_file = TEXT
					tx_solr_file {
						typolink {
							parameter.field = fileRelativePath
							returnLast = url
						}
					}
				}

				link = CASE
				link {
					tx_solr_file = TEXT
					tx_solr_file {
						field = fileName

						typolink {
							parameter.field = fileRelativePath
						}
					}
				}

			}

		}

	}

}
